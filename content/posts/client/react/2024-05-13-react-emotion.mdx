```
import {css} from "@emotion/react";
function Test() {
    return(<img
                css={css`width:100%;
                border-radius: "2px";
                background-color: #FEE500;
                cursor: pointer;
             `}
            />)
}
export default Test
```

css() 함수 를 써서 style 속성과 유사하게 작성 가능

# props 로 넘어온 값 추가하기 or css 함수 작성 후 추가
```
import {css} from "@emotion/react";
function Test({backColor}) {
    return(<img
                css={css`width:100%;
                border-radius: "2px";
                background-color: ${backColor};
                cursor: pointer;
             `}
            />)
}
export default Test
```
